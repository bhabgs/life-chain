### 4.1 对话陪伴系统

#### 技术方案
```
用户输入 → 预处理 → 上下文检索 → LLM推理 → 后处理 → 返回结果
   ↓          ↓          ↓           ↓         ↓
 情绪识别   意图识别   记忆检索    流式输出   内容审核
```

**实现细节**：
1. **上下文管理**：
   - 短期记忆：最近N轮对话（Redis存储）
   - 长期记忆：向量检索相关记忆（Milvus）
   - 人格记忆：固定的人格设定

2. **Prompt工程**：
```typescript
const buildPrompt = (user, personality, context, memories) => {
  return `
你是${user.name}的数字人格伴侣，你们的关系是${personality.relationship}。

## 你的性格特征
${personality.traits}

## 最近的对话
${context.recentChat}

## 相关记忆
${memories.map(m => m.content).join('\n')}

## 当前对话
用户：${context.currentMessage}
`;
};
```

3. **流式输出**：
```typescript
// 使用SSE返回流式内容
async function* streamChat(prompt: string) {
  const stream = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [{ role: 'user', content: prompt }],
    stream: true,
  });

  for await (const chunk of stream) {
    yield chunk.choices[0]?.delta?.content || '';
  }
}
```

#### 记忆检索算法

```typescript
// 混合检索：关键词 + 语义相似度
async function searchMemories(userId: string, query: string, options: SearchOptions) {
  // 1. 关键词检索（ElasticSearch）
  const keywordResults = await es.search({
    index: 'memories',
    body: {
      query: {
        bool: {
          must: [
            { match: { user_id: userId } },
            { multi_match: { query, fields: ['content', 'metadata.tags'] } }
          ]
        }
      }
    }
  });

  // 2. 语义检索（Milvus）
  const queryVector = await getEmbedding(query);
  const vectorResults = await milvus.search({
    collection_name: 'memory_vectors',
    vectors: [queryVector],
    filter: `user_id == "${userId}"`,
    top_k: 20
  });

  // 3. 结果融合（RRF算法）
  const fusedResults = reciprocalRankFusion([
    keywordResults.hits.hits,
    vectorResults
  ]);

  // 4. 重排序（根据时间、重要性等）
  return rerank(fusedResults, options);
}
```

```typescript
// 多模态融合
function fuseEmotions(textEmotion: EmotionResult, voiceEmotion: EmotionResult) {
  const weights = { text: 0.6, voice: 0.4 };

  return {
    emotion: weightedVote([textEmotion, voiceEmotion], weights),

                 voiceEmotion.confidence * weights.voice)
```

```typescript
// 人格模型（基于五大人格理论 + 自定义维度）
interface PersonalityModel {
  // 五大人格维度
  bigFive: {
    openness: number;        // 开放性 (0-1)
    conscientiousness: number; // 尽责性
    extraversion: number;    // 外向性
    agreeableness: number;   // 宜人性
    neuroticism: number;     // 神经质
  };

  // 自定义维度
  custom: {
    warmth: number;          // 温暖度
    rationality: number;     // 理性度
    humorousness: number;    // 幽默感
    formality: number;       // 正式度
  };

  // 行为风格
  behaviorStyle: {
    responseLength: 'short' | 'medium' | 'long';
    emojiUsage: 'none' | 'rare' | 'frequent';
    questionFrequency: number; // 提问频率
  };

  // 版本控制
  version: string;
  createdAt: Date;
  updatedAt: Date;
}

// 人格演进算法
class PersonalityEvolution {
  // 基于互动数据微调人格
  async evolve(personality: PersonalityModel, interactions: Interaction[]) {
    // 1. 分析互动模式
    const patterns = await this.analyzeInteractionPatterns(interactions);

    // 2. 计算人格调整量（小步调整，保持核心稳定）
    const adjustments = this.calculateAdjustments(patterns);

    // 3. 应用约束（防止偏移过大）
    const constrainedAdjustments = this.applyConstraints(adjustments, {
      maxChange: 0.05, // 单次最大变化5%
      coreLock: true   // 锁定核心特质
    });

    // 4. 生成新版本
    return this.applyAdjustments(personality, constrainedAdjustments);
  }
}
```

```typescript
async function verifyHeirAccess(heirId: string, deceasedId: string) {
  // 1. 验证继承人身份
  const heir = await db.heir.findUnique({
    where: { id: heirId, deceasedId }
  });

  if (!heir) throw new UnauthorizedError();

  // 2. 验证继承条件（可能需要死亡证明等）
  const conditions = await verifyConditions(heir);

    canExport: heir.permissions.includes('export'),
```

```typescript
  // 1. 文本审核
  const textResult = await textModerator.check(content.text);

  // 2. 图片审核
  const imageResult = content.images ?
    await imageModerator.check(content.images) : null;
```
